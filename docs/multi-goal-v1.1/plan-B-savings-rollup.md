## JavaScript: the rollup helper

In a new section of `js/calc-income.js` (or new `js/calc-savings-rollup.js` if it grows; default: keep in calc-income.js for now):

```js
/* ---------- Current Savings rollup (Pardha v1.1 §3.4) ----------
 * Total = seed + tracker contributions compounded monthly at post-tax blended return.
 * Updates the readonly #currentSavings field which downstream code reads via RP.val.
 */

RP._computeSavingsRollup = function () {
    const seedEl = document.getElementById('currentSavingsSeed');
    const rollupEl = document.getElementById('trackerRollupAmount');
    const totalEl = document.getElementById('currentSavings');
    if (!seedEl || !rollupEl || !totalEl) return; // tab not in DOM yet

    const seed = parseFloat(seedEl.value) || 0;

    // Compute tracker rollup with monthly compound interest
    const trackerEntries = (typeof RP._tracker !== 'undefined' && Array.isArray(RP._tracker.entries))
        ? RP._tracker.entries
        : [];
    const blendedReturn = (typeof RP._blendedReturn === 'number' && isFinite(RP._blendedReturn))
        ? RP._blendedReturn
        : (typeof RP._preReturn === 'number' && isFinite(RP._preReturn) ? RP._preReturn : 0.10);
    const taxRate = (typeof RP._taxRate === 'number' && isFinite(RP._taxRate))
        ? RP._taxRate
        : (RP.val ? RP.val('taxRate') / 100 : 0.30);

    const postTaxAnnual = blendedReturn * (1 - taxRate);
    const monthlyRate = Math.pow(1 + postTaxAnnual, 1/12) - 1;

    const today = new Date();
    let trackerRollup = 0;

    trackerEntries.forEach(entry => {
        const amount = parseFloat(entry.amount) || 0;
        if (!entry.date) {
            trackerRollup += amount; // no date → assume now, no interest yet
            return;
        }
        const entryDate = new Date(entry.date);
        if (isNaN(entryDate.getTime())) {
            trackerRollup += amount;
            return;
        }
        const monthsElapsed = Math.max(0,
            (today.getFullYear() - entryDate.getFullYear()) * 12
            + (today.getMonth() - entryDate.getMonth())
        );
        const compounded = amount * Math.pow(1 + monthlyRate, monthsElapsed);
        trackerRollup += compounded;
    });

    // Round to nearest rupee for display (numerical precision is fine internally)
    trackerRollup = Math.round(trackerRollup);

    rollupEl.value = trackerRollup;
    totalEl.value = seed + trackerRollup;

    // Trigger downstream readers (Projections, What-If, Multi-Goal renderers)
    totalEl.dispatchEvent(new Event('change', { bubbles: true }));
};

// Wire change listeners
function setupSavingsRollup() {
    const seedEl = document.getElementById('currentSavingsSeed');
    if (seedEl) {
        seedEl.addEventListener('input', RP._computeSavingsRollup);
    }
    // Also re-compute when Financial Plan's blended return updates
    // (existing code likely fires a custom event or just sets RP._blendedReturn — listen to whichever)
    const taxEl = document.getElementById('taxRate');
    if (taxEl) taxEl.addEventListener('input', RP._computeSavingsRollup);

    // Initial compute on load (after Financial Plan + Tracker have populated their state)
    setTimeout(RP._computeSavingsRollup, 200);
}

if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', setupSavingsRollup);
} else {
    setupSavingsRollup();
}
```

## Risks

- **Tracker schema may not have a `date` field** in v1 — check `js/calc-tracker.js`. If entries are dateless, ALL contributions get treated as "now" → rollup = sum without interest. Acceptable behavior for v1.1; future tracker updates can add date fields.
- **`RP._blendedReturn` may not be a global** — calc-financial-plan.js may compute it inline. Convention-checker resolves the exact name. Fallback: compute it ourselves from existing input fields.
- **Tax rate location** — `taxRate` input is on Basics tab as a percentage (e.g., 30 not 0.30). Helper divides by 100. Make sure this matches what `calc-financial-plan.js` does internally.
- **Interest computed on stale Financial Plan values** — if user changes Financial Plan inputs AFTER tracker entries were logged, the rollup recomputes with new rates. Correct behavior (interest is forward-looking from the date), but worth noting in a comment.
- **Multi-Goal renderer race**: `_readCorpusAtRetirement` calls `RP.calculateProjections()` defensively (commit 4c7e93f). With Feature B, currentSavings becomes computed — make sure `RP._computeSavingsRollup` runs BEFORE `calculateProjections` reads it. Order: load page → setupSavingsRollup → setTimeout(200ms compute) → calculateProjections → multigoal renders. Should be fine; verify in cold-start playwright.
